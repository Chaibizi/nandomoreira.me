A Estrutura base para dar o pontapé inicial com HTML5.

{% highlight html %}
<!DOCTYPE html>
<html lang="pt-br">
  <head>
    <meta charset="utf-8">
    <link rel="stylesheet" type="text/css" href="/main.css" />
    <title>Titulo do seu projeto</title>
  </head>
  <body>
    <main>
      <p>Estrutura básica de um HTML</p>
    </main>
  </body>
</html>
{% endhighlight %}
